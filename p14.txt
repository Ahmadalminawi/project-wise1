        // جلب البيانات من localStorage وعرضها
        window.onload = function () {
            let tenantData = JSON.parse(localStorage.getItem("tenantData"));
            
            if (tenantData) {
                document.getElementById("fullNameTenant").innerText = tenantData.fullNameTenant;
                document.getElementById("age").innerText = tenantData.age;
                document.getElementById("maritalStatus").innerText = tenantData.maritalStatus;
                document.getElementById("nationality").innerText = tenantData.nationality;
                document.getElementById("identityPhoto").innerText = tenantData.identityPhoto;
                document.getElementById("tenantPhoto").innerText = tenantData.tenantPhoto;

                // عرض الصورة الشخصية للمستأجر
                let profileImage = document.getElementById("profileImage");
                profileImage.src = URL.createObjectURL(tenantData.tenantPhoto); // تحديث الصورة
            }
        };
